22  /*
23  * This function calculates the curvature of the current level set function being
24  * the solution of an intermediate projection step
25  *
26  * (w, κ) + η_κ (∇w, ∇κ) = (w,∇·n_ϕ)
27  * Ω Ω Ω
28  *
29  * with test function w, curvature κ, damping parameter η_κ and the normal to the
30  * level set function n_ϕ.
31  *
32  */
33  template<int dim, unsigned int comp=0, typename number = double>
34  class CurvatureOperator: public OperatorBase<number,
35  LinearAlgebra::distributed::Vector<number>,
36  LinearAlgebra::distributed::BlockVector<number>>
37  {
38  public:

53  void
54  assemble_matrixbased( const BlockVectorType & solution_normal_vector_in,
55  SparseMatrixType & matrix,
56  VectorType & rhs ) const override
57  {
58  solution_normal_vector_in.update_ghost_values();
59 
60  const auto& mapping = scratch_data.get_mapping();
61  FEValues<dim> fe_values( mapping,
62  scratch_data.get_matrix_free().get_dof_handler(comp).get_fe(),
63  scratch_data.get_matrix_free().get_quadrature(comp),
64  update_values | update_gradients | update_quadrature_points | update_JxW_values );
65 
66  const unsigned int dofs_per_cell =scratch_data.get_n_dofs_per_cell(comp);
67 
68  FullMatrix<double> curvature_cell_matrix( dofs_per_cell, dofs_per_cell );
69  Vector<double> curvature_cell_rhs( dofs_per_cell) ;
70  std::vector<types::global_dof_index> local_dof_indices( dofs_per_cell );
71 
72  const unsigned int n_q_points = fe_values.get_quadrature().size();
73 
74  std::vector<Tensor<1,dim>> normal_at_q( n_q_points, Tensor<1,dim>() );
75 
76  matrix = 0.0;
77  rhs = 0.0;
78 
79  for (const auto &cell : scratch_data.get_matrix_free().get_dof_handler(comp).active_cell_iterators())
80  if (cell->is_locally_owned())
81  {
82  fe_values.reinit(cell);
83  cell->get_dof_indices( local_dof_indices );
84 
85  curvature_cell_matrix = 0.0;
86  curvature_cell_rhs = 0.0;
87 
88  NormalVector::NormalVectorOperator<dim,comp>::get_unit_normals_at_quadrature( fe_values,
89  solution_normal_vector_in,
90  normal_at_q );
91 
92  for (const unsigned int q_index : fe_values.quadrature_point_indices())
93  {
94  for (unsigned int i=0; i<dofs_per_cell; ++i)
95  {
96  const double phi_i = fe_values.shape_value( i, q_index );
97  const Tensor<1,dim> grad_phi_i = fe_values.shape_grad( i, q_index );
98 
99  for (unsigned int j=0; j<dofs_per_cell; ++j)
100  {
101  const double phi_j = fe_values.shape_value( j, q_index);
102  const Tensor<1,dim> grad_phi_j = fe_values.shape_grad( j, q_index);
103 
104  curvature_cell_matrix( i, j ) += ( phi_i * phi_j
105  +
106  damping * grad_phi_i * grad_phi_j
107  )
108  *
109  fe_values.JxW( q_index ) ;
110  }
111  curvature_cell_rhs(i) += ( grad_phi_i
112  *
113  normal_at_q[ q_index ]
114  *
115  fe_values.JxW( q_index ) );
116  }
117  }
118 
119  //assembly
120  cell->get_dof_indices(local_dof_indices);
121  scratch_data.get_constraint(comp).distribute_local_to_global( curvature_cell_matrix,
122  curvature_cell_rhs,
123  local_dof_indices,
124  matrix,
125  rhs);
126 
127  } // end of cell loop
128  matrix.compress( VectorOperation::add );
129  rhs.compress( VectorOperation::add );
130  }

136  void
137  vmult(VectorType & dst,
138  const VectorType & src) const override
139  {
140 
141  scratch_data.get_matrix_free().template cell_loop<VectorType, VectorType>( [&]
142  (const auto&, auto& dst, const auto& src, auto cell_range) {
143  FECellIntegrator<dim, 1, number> curvature( scratch_data.get_matrix_free(), comp, comp);
144  for (unsigned int cell = cell_range.first; cell < cell_range.second; ++cell)
145  {
146  curvature.reinit(cell);
147  curvature.gather_evaluate(src, true, true);
148 
149  for (unsigned int q_index=0; q_index<curvature.n_q_points; ++q_index)
150  {
151  curvature.submit_value( curvature.get_value( q_index ), q_index);
152  curvature.submit_gradient( damping * curvature.get_gradient( q_index ), q_index );
153  }
154 
155  curvature.integrate_scatter(true, true, dst);
156  }
157  },
158  dst,
159  src,
160  true );
161  }

163  void
164  create_rhs(VectorType & dst,
165  const BlockVectorType & src) const override
166  {
167  scratch_data.get_matrix_free().template cell_loop<VectorType, BlockVectorType>(
168  [&](const auto &, auto &dst, const auto &src, auto macro_cells) {
169  FECellIntegrator<dim, 1, number> curvature( scratch_data.get_matrix_free(), comp, comp);
170  FECellIntegrator<dim, dim, number> normal_vector( scratch_data.get_matrix_free(), comp, comp);
171  for (unsigned int cell = macro_cells.first; cell < macro_cells.second; ++cell)
172  {
173  curvature.reinit(cell);
174 
175  normal_vector.reinit(cell);
176  normal_vector.read_dof_values_plain(src);
177  normal_vector.evaluate(true, false);
178 
179  for (unsigned int q_index = 0; q_index < curvature.n_q_points; ++q_index)
180  {
181  const auto n_phi = Reinitialization::OlssonOperator<dim,comp>::normalize(normal_vector.get_value(q_index));
182  curvature.submit_gradient( n_phi, q_index );
183  }
184 
185  curvature.integrate_scatter(false, true, dst);
186  }
187  },
188  dst,
189  src,
190  true);
191  }
