51  void
52  assemble_matrixbased( const VectorType & levelset_old,
53  SparseMatrixType & matrix,
54  VectorType & rhs ) const override
55  {
56 
57  levelset_old.update_ghost_values();
58 
59  FEValues<dim> fe_values( scratch_data.get_mapping(),
60  scratch_data.get_matrix_free().get_dof_handler(comp).get_fe(),
61  scratch_data.get_matrix_free().get_quadrature(comp),
62  update_values | update_gradients | update_quadrature_points | update_JxW_values
63  );
64  const unsigned int dofs_per_cell =scratch_data.get_n_dofs_per_cell();
65 
66  FullMatrix<double> cell_matrix( dofs_per_cell, dofs_per_cell );
67  Vector<double> cell_rhs( dofs_per_cell );
68 
69  const unsigned int n_q_points = fe_values.get_quadrature().size();
70 
71  std::vector<double> psi_at_q( n_q_points );
72  std::vector<Tensor<1,dim>> grad_psi_at_q( n_q_points, Tensor<1,dim>() );
73  std::vector<Tensor<1,dim>> normal_at_q( n_q_points, Tensor<1,dim>() );
74 
75  std::vector<types::global_dof_index> local_dof_indices( dofs_per_cell );
76 
77  rhs = 0.0;
78  matrix = 0.0;
79 
80  this->normal_vec.update_ghost_values();
81 
82  for (const auto &cell : scratch_data.get_matrix_free().get_dof_handler(comp).active_cell_iterators())
83  if (cell->is_locally_owned())
84  {
85  cell_matrix = 0.0;
86  cell_rhs = 0.0;
87  fe_values.reinit(cell);
88 
89  const double epsilon_cell = eps>0.0 ? eps : cell->diameter() / ( std::sqrt(dim) ) * eps_scale_factor;
90  AssertThrow(epsilon_cell>0.0, ExcMessage("Reinitialization: the value of epsilon for the reinitialization function must be larger than zero!"));
91 
92  fe_values.get_function_values( levelset_old, psi_at_q ); // compute values of old solution at tau_n
93  fe_values.get_function_gradients( levelset_old, grad_psi_at_q ); // compute gradients of old solution at tau_n
94  NormalVector::NormalVectorOperator<dim,comp>::get_unit_normals_at_quadrature(fe_values,
95  this->normal_vec,
96  normal_at_q);
97 
98  for (const unsigned int q_index : fe_values.quadrature_point_indices())
99  {
100  for (const unsigned int i : fe_values.dof_indices())
101  {
102  const double nTimesGradient_i = normal_at_q[q_index] * fe_values.shape_grad(i, q_index);
103 
104  for (const unsigned int j : fe_values.dof_indices())
105  {
106  const double nTimesGradient_j = normal_at_q[q_index] * fe_values.shape_grad(j, q_index);
107  //clang-format off
108  cell_matrix(i,j) += (
109  fe_values.shape_value(i,q_index) * fe_values.shape_value(j,q_index)
110  +
111  this->d_tau * epsilon_cell * nTimesGradient_i * nTimesGradient_j
112  )
113  *
114  fe_values.JxW( q_index );
115  //clang-format on
116  }
117 
118  const double diffRhs = epsilon_cell * normal_at_q[q_index] * grad_psi_at_q[q_index];
119 
120  //clang-format off
121  const auto compressive_flux = [](const double psi) { return 0.5 * ( 1. - psi * psi ); };
122  cell_rhs(i) += ( compressive_flux(psi_at_q[q_index]) - diffRhs )
123  *
124  nTimesGradient_i
125  *
126  this->d_tau
127  *
128  fe_values.JxW( q_index );
129  //clang-format on
130  }
131  } //end loop over gauss points
132  // assembly
133  cell->get_dof_indices( local_dof_indices );
134 
135  scratch_data.get_constraint(comp).distribute_local_to_global( cell_matrix,
136  cell_rhs,
137  local_dof_indices,
138  matrix,
139  rhs);
140 
141  }
142 
143  matrix.compress( VectorOperation::add );
144  rhs.compress( VectorOperation::add );
145 
146  }

153  void
154  vmult(VectorType & dst,
155  const VectorType & src) const override
156  {
157  AssertThrow(this->d_tau>0.0, ExcMessage("reinitialization operator: d_tau must be set"));
158 
159  const double eps_ = eps > 0 ? eps : eps_scale_factor * scratch_data.get_min_cell_size(comp); // @ todo: check how cell size can be extracted from matrix free class
160 
161  AssertThrow(eps_>0.0, ExcMessage("reinitialization operator: epsilon must be set"));
162 
163 
164  scratch_data.get_matrix_free().template cell_loop<VectorType, VectorType>( [&]
165  (const auto&, auto& dst, const auto& src, auto cell_range) {
166  FECellIntegrator<dim, 1, number> levelset( scratch_data.get_matrix_free(),comp,comp,comp );
167  FECellIntegrator<dim, dim, number> normal_vector( scratch_data.get_matrix_free(),comp,comp,comp );
168  for (unsigned int cell = cell_range.first; cell < cell_range.second; ++cell)
169  {
170  levelset.reinit(cell);
171  levelset.gather_evaluate(src, true, true);
172 
173  normal_vector.reinit(cell);
174  normal_vector.read_dof_values_plain(this->normal_vec);
175  normal_vector.evaluate(true, false);
176 
177  for (unsigned int q_index=0; q_index<levelset.n_q_points; q_index++)
178  {
179  const scalar phi = levelset.get_value( q_index );
180 
181  const vector grad_phi = levelset.get_gradient( q_index );
182 
183  const auto n_phi = normalize(normal_vector.get_value(q_index));
184 
185  levelset.submit_value(phi, q_index);
186  levelset.submit_gradient(this->d_tau * eps_ * scalar_product(grad_phi, n_phi) * n_phi, q_index);
187  }
188 
189  levelset.integrate_scatter(true, true, dst);
190  }
191  },
192  dst,
193  src,
194  true );

197  void
198  create_rhs(VectorType & dst,
199  const VectorType & src) const override
200  {
201  this->normal_vec.update_ghost_values();
202 
203  AssertThrow(this->d_tau>0.0, ExcMessage("reinitialization matrix-free operator: d_tau must be set"));
204  const double eps_ = eps > 0 ? eps : eps_scale_factor * scratch_data.get_min_cell_size(comp); // @ todo: check how cell size can be extracted from matrix free class
205 
206  AssertThrow(eps_>0.0, ExcMessage("reinitialization matrix-free operator: epsilon must be set"));
207 
208  const auto compressive_flux = [&](const auto &phi)
209  {
210  return 0.5 * ( make_vectorized_array<number>(1.) - phi * phi );
211  };
212 
213 
214  scratch_data.get_matrix_free().template cell_loop<VectorType, VectorType>(
215  [&](const auto &, auto &dst, const auto &src, auto macro_cells) {
216  FECellIntegrator<dim, 1, number> psi( scratch_data.get_matrix_free(),comp,comp,0);
217  FECellIntegrator<dim, dim, number> normal_vector( scratch_data.get_matrix_free(),comp,comp,0);
218  for (unsigned int cell = macro_cells.first; cell < macro_cells.second; ++cell)
219  {
220  psi.reinit(cell);
221  psi.gather_evaluate(src, true, true);
222 
223  normal_vector.reinit(cell);
224  normal_vector.read_dof_values_plain(this->normal_vec);
225  normal_vector.evaluate(true, false);
226 
227  for (unsigned int q_index = 0; q_index < psi.n_q_points; ++q_index)
228  {
229  const scalar val = psi.get_value(q_index);
230  const auto n_phi = normalize(normal_vector.get_value(q_index));
231 
232  psi.submit_gradient( this->d_tau * compressive_flux(val) * n_phi
233  -
234  this->d_tau * eps_ * scalar_product( psi.get_gradient(q_index), n_phi ) * n_phi, q_index);
235  }
236 
237  psi.integrate_scatter(false, true, dst);
238  }
239  },
240  dst,
241  src,
242  true);
243  }
