25  * Level set model including advection, reinitialization and curvature computation

27  */

43  void
44  initialize(const std::shared_ptr<const ScratchData<dim>> &scratch_data_in,
45  const VectorType & solution_level_set_in,
46  const Parameters<double> & data_in,
47  const unsigned int dof_idx_in,
48  const unsigned int dof_no_bc_idx_in,
49  const unsigned int quad_idx_in)
50  {
51  scratch_data = scratch_data_in;
52  dof_idx = dof_idx_in;
53  /*
54  * set the level set data
55  */
56  level_set_data = data_in.ls;
57  /*
58  * initialize the advection_diffusion problem
59  */
60  advec_diff_operation.initialize(
61  scratch_data, solution_level_set_in, data_in, dof_idx, dof_no_bc_idx_in, quad_idx_in);
62  /*
63  * set the parameters for the levelset problem; already determined parameters
64  * from the initialize call of advec_diff_operation are overwritten.
65  */
66  set_level_set_parameters(data_in);
67  /*
68  * initialize the reinitialization operation class
69  */
70  reinit_operation.initialize(
71  scratch_data, solution_level_set_in, data_in, dof_no_bc_idx_in, quad_idx_in);
72  /*
73  * The initial solution of the level set equation will be reinitialized.
74  */
75  if (level_set_data.do_reinitialization)
76  {
77  while (!reinit_time_iterator.is_finished())
78  {
79  const double d_tau = reinit_time_iterator.get_next_time_increment();
80  scratch_data->get_pcout() << std::setw(4) << ""
81  << "| reini: τ= " << std::setw(10) << std::left
82  << reinit_time_iterator.get_current_time();
83  reinit_operation.solve(d_tau);
84  }
85  advec_diff_operation.solution_advected_field =
86  reinit_operation.solution_level_set; // @ could be defined by reference
87  reinit_time_iterator.reset();
88  }
89  /*
90  * initialize the curvature operation class
91  */
92  curvature_operation.initialize(scratch_data, data_in, dof_no_bc_idx_in, quad_idx_in);
93  }
94 
95  void
96  solve(const double dt, const BlockVectorType &advection_velocity)
97  {
98  /*
99  * solve the advection step of the level set function
100  */
101  advec_diff_operation.solve(dt, advection_velocity);
102  /*
103  * solve the reinitialization problem of the level set equation
104  */
105  if (level_set_data.do_reinitialization)
106  {
107  reinit_operation.update_initial_solution(advec_diff_operation.solution_advected_field);
108 
109  while (!reinit_time_iterator.is_finished())
110  {
111  const double d_tau = reinit_time_iterator.get_next_time_increment();
112  scratch_data->get_pcout() << std::setw(4) << ""
113  << "| reini: τ= " << std::setw(10) << std::left
114  << reinit_time_iterator.get_current_time();
115  reinit_operation.solve(d_tau);
116  }
117 
118  /*
119  * reset the solution of the level set field to the reinitialized solution
120  */
121  advec_diff_operation.solution_advected_field =
122  reinit_operation.solution_level_set; // @ could be defined by reference
123  reinit_time_iterator.reset();
124  }
125  /*
126  * compute the curvature
127  */
128  curvature_operation.solve(advec_diff_operation.solution_advected_field);
129  }

132  void
133  set_level_set_parameters(const Parameters<double> &data_in)
134  {
135  level_set_data.do_reinitialization = data_in.ls.do_reinitialization;
136  advec_diff_operation.advec_diff_data.diffusivity = data_in.ls.artificial_diffusivity;
137  advec_diff_operation.advec_diff_data.theta = data_in.ls.theta;
138  advec_diff_operation.advec_diff_data.do_print_l2norm = data_in.ls.do_print_l2norm;
139  advec_diff_operation.advec_diff_data.do_matrix_free = data_in.ls.do_matrix_free;
140  /*
141  * setup the time iterator for the reinitialization problem
142  */
143  reinit_time_iterator.initialize(
144  TimeIteratorData<double>{0.0,
145  100000.,
146  data_in.reinit.dtau > 0.0 ?
147  data_in.reinit.dtau :
148  scratch_data->get_min_cell_size(dof_idx) *
149  data_in.reinit.scale_factor_epsilon,
150  data_in.reinit.max_n_steps,
151  false});
152  }
