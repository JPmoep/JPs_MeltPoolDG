25  * Level set model including advection, reinitialization and curvature computation

42  void
43  initialize(const std::shared_ptr<const ScratchData<dim>> &scratch_data_in,
44  const VectorType & solution_level_set_in,
45  const Parameters<double>& data_in,
46  const std::shared_ptr<TensorFunction<1,dim>> &advection_velocity_in )
47  {
48  scratch_data = scratch_data_in;
49  advection_velocity = advection_velocity_in;
50  /*
51  * set the level set data
52  */
53  level_set_data = data_in.ls;
54  /*
55  * initialize the advection_diffusion problem
56  */
57  advec_diff_operation.initialize(scratch_data,
58  solution_level_set_in,
59  data_in,
60  *advection_velocity_in);
61  /*
62  * set the parameters for the levelset problem; already determined parameters
63  * from the initialize call of advec_diff_operation are overwritten.
64  */
65  set_level_set_parameters(data_in);
66  /*
67  * initialize the reinitialization operation class
68  */
69  reinit_operation.initialize(scratch_data,
70  solution_level_set_in,
71  data_in);
72  /*
73  * The initial solution of the level set equation will be reinitialized.
74  */
75  if(level_set_data.do_reinitialization)
76  {
77  while ( !reinit_time_iterator.is_finished() )
78  {
79  const double d_tau = reinit_time_iterator.get_next_time_increment();
80  scratch_data->get_pcout() << std::setw(4) << "" << "| reini: τ= " << std::setw(10) << std::left << reinit_time_iterator.get_current_time();
81  reinit_operation.solve(d_tau);
82  }
83  advec_diff_operation.solution_advected_field = reinit_operation.solution_level_set; // @ could be defined by reference
84  reinit_time_iterator.reset();
85  }
86  /*
87  * initialize the curvature operation class
88  */
89  curvature_operation.initialize(scratch_data, data_in);
90  }
91 
92  void
93  solve(const double dt)
94  {
95  /*
96  * solve the advection step of the level set function
97  */
98  advec_diff_operation.solve( dt );
99  /*
100  * solve the reinitialization problem of the level set equation
101  */
102  if(level_set_data.do_reinitialization)
103  {
104  reinit_operation.update_initial_solution(advec_diff_operation.solution_advected_field);
105 
106  while ( !reinit_time_iterator.is_finished() )
107  {
108  const double d_tau = reinit_time_iterator.get_next_time_increment();
109  scratch_data->get_pcout() << std::setw(4) << "" << "| reini: τ= " << std::setw(10) << std::left << reinit_time_iterator.get_current_time();
110  reinit_operation.solve(d_tau);
111  }
112 
113  /*
114  * reset the solution of the level set field to the reinitialized solution
115  */
116  advec_diff_operation.solution_advected_field = reinit_operation.solution_level_set; // @ could be defined by reference
117  reinit_time_iterator.reset();
118  }
119  /*
120  * compute the curvature
121  */
122  curvature_operation.solve(advec_diff_operation.solution_advected_field);
123  }

126  void
127  set_level_set_parameters(const Parameters<double>& data_in)
128  {
129  level_set_data.do_reinitialization = data_in.ls.do_reinitialization;
130  advec_diff_operation.advec_diff_data.diffusivity = data_in.ls.artificial_diffusivity;
131  advec_diff_operation.advec_diff_data.theta = data_in.ls.theta;
132  advec_diff_operation.advec_diff_data.do_print_l2norm = data_in.ls.do_print_l2norm;
133  advec_diff_operation.advec_diff_data.do_matrix_free = data_in.ls.do_matrix_free;
134  /*
135  * setup the time iterator for the reinitialization problem
136  */
137  reinit_time_iterator.initialize(TimeIteratorData<double>{0.0,
138  100000.,
139  data_in.reinit.dtau > 0.0 ? data_in.reinit.dtau : scratch_data->get_min_cell_size() * data_in.reinit.scale_factor_epsilon,
140  data_in.reinit.max_n_steps,
141  false});
142 
143  }
