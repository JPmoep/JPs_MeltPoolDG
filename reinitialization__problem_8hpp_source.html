85  /*
86  * This function initials the relevant member data

88  */
89  void
90  initialize(std::shared_ptr<SimulationBase<dim>> base_in)
91  {
92  /*
93  * initialize the dof system
94  */
95  setup_dof_system(base_in, true);
96  /*
97  * initialize the time iterator
98  */
99  time_iterator.initialize(TimeIteratorData<double>{0.0,
100  10000.,
101  base_in->parameters.reinit.dtau,
102  base_in->parameters.reinit.max_n_steps,
103  false});
104  /*
105  * set initial conditions of the levelset function
106  */
107  VectorType solution_level_set;
108  scratch_data->initialize_dof_vector(solution_level_set);
109  VectorTools::project(dof_handler,
110  constraints,
111  scratch_data->get_quadrature(),
112  *base_in->get_field_conditions()->initial_field,
113  solution_level_set);
114 
115  solution_level_set.update_ghost_values();
116 
117  /*
118  * initialize the reinitialization operation class
119  */
120  reinit_operation.initialize(scratch_data, solution_level_set, base_in->parameters);
121  }

123  void
124  setup_dof_system(std::shared_ptr<SimulationBase<dim>> base_in, const bool do_initial_setup)
125  {
126  /*
127  * setup scratch data
128  */
129  if (do_initial_setup)
130  {
131  scratch_data = std::make_shared<ScratchData<dim>>();
132  /*
133  * setup mapping
134  */
135  const auto mapping = MappingQGeneric<dim>(base_in->parameters.base.degree);
136  scratch_data->set_mapping(mapping);
137  /*
138  * create quadrature rule
139  */
140  QGauss<1> quad_1d_temp(base_in->parameters.base.n_q_points_1d);
141  scratch_data->attach_quadrature(quad_1d_temp);
142  }
143  /*
144  * setup DoFHandler
145  */
146  FE_Q<dim> fe(base_in->parameters.base.degree);
147  dof_handler.initialize(*base_in->triangulation, fe);
148 
149  if (do_initial_setup)
150  scratch_data->attach_dof_handler(dof_handler);
151  /*
152  * re-create partitioning
153  */
154  scratch_data->create_partitioning();
155  /*
156  * make hanging nodes constraints
157  */
158  constraints.clear();
159  constraints.reinit(scratch_data->get_locally_relevant_dofs());
160  DoFTools::make_hanging_node_constraints(dof_handler, constraints);
161  constraints.close();
162 
163  if (do_initial_setup)
164  scratch_data->attach_constraint_matrix(constraints);
165 
166  /*
167  * create the matrix-free object
168  */
169  scratch_data->build();
170  }

172  /*
173  * perform mesh refinement
174  */
175  void
176  refine_mesh(std::shared_ptr<SimulationBase<dim>> base_in)
177  {
178  if (auto tria = std::dynamic_pointer_cast<parallel::distributed::Triangulation<dim>>(
179  base_in->triangulation))
180  {
181  Vector<float> estimated_error_per_cell(base_in->triangulation->n_active_cells());
182 
183  /* @todo:
184  * bug (?)
185  * for the purpose of the KellyErrorEstimator initialize_dof_vector could not be used
186  * scratch_data->initialize_dof_vector(locally_relevant_solution);
187  */
188 
189  VectorType locally_relevant_solution;
190  locally_relevant_solution.reinit(scratch_data->get_partitioner());
191  locally_relevant_solution.copy_locally_owned_data_from(
192  reinit_operation.solution_level_set);
193  constraints.distribute(locally_relevant_solution);
194  locally_relevant_solution.update_ghost_values();
195 
196  KellyErrorEstimator<dim>::estimate(scratch_data->get_dof_handler(),
197  scratch_data->get_face_quadrature(),
198  {},
199  locally_relevant_solution,
200  estimated_error_per_cell);
201 
202  parallel::distributed::GridRefinement::refine_and_coarsen_fixed_fraction(
203  *tria,
204  estimated_error_per_cell,
205  base_in->parameters.amr.upper_perc_to_refine,
206  base_in->parameters.amr.lower_perc_to_coarsen);
207 
208  /*
209  * Limit the maximum and minimum refinement levels of cells of the grid.
210  */
211  if (tria->n_levels() > base_in->parameters.amr.max_grid_refinement_level)
212  for (auto &cell : tria->active_cell_iterators_on_level(
213  base_in->parameters.amr.max_grid_refinement_level))
214  cell->clear_refine_flag();
215  if (tria->n_levels() < base_in->parameters.amr.min_grid_refinement_level)
216  for (auto &cell : tria->active_cell_iterators_on_level(
217  base_in->parameters.amr.min_grid_refinement_level))
218  cell->clear_coarsen_flag();
219 
220  /*
221  * Initialize the triangulation change from the old grid to the new grid
222  */
223  base_in->triangulation->prepare_coarsening_and_refinement();
224  /*
225  * Initialize the solution transfer from the old grid to the new grid
226  */
227  parallel::distributed::SolutionTransfer<dim, VectorType, DoFHandlerType>
228  solution_transfer(dof_handler);
229  solution_transfer.prepare_for_coarsening_and_refinement(locally_relevant_solution);
230 
231  /*
232  * Execute the grid refinement
233  */
234  base_in->triangulation->execute_coarsening_and_refinement();
235 
236  /*
237  * update dof-related scratch data to match the current triangulation
238  */
239  setup_dof_system(base_in, false);
240 
241  /*
242  * interpolate the given solution to the new discretization
243  *
244  */
245  VectorType interpolated_solution;
246  scratch_data->initialize_dof_vector(interpolated_solution);
247 
248  solution_transfer.interpolate(interpolated_solution);
249 
250  constraints.distribute(interpolated_solution);
251  /*
252  * update the reinitialization operator with the new solution values
253  */
254  reinit_operation.update_initial_solution(interpolated_solution);
255  }
256  else
257  //@todo: WIP
258  AssertThrow(false, ExcMessage("Mesh refinement for dim=1 not yet supported"));
259  }
