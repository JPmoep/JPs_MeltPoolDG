55  void
56  assemble_matrixbased(const VectorType &advected_field_old,
57  SparseMatrixType &matrix,
58  VectorType & rhs) const override
59  {
60  AssertThrow(data.diffusivity >= 0.0,
61  ExcMessage("Advection diffusion operator: diffusivity is smaller than zero!"));
62 
63  advected_field_old.update_ghost_values();
64 
65 
66  FEValues<dim> fe_values(scratch_data.get_mapping(),
67  scratch_data.get_dof_handler(this->dof_idx).get_fe(),
68  scratch_data.get_quadrature(this->quad_idx),
69  update_values | update_gradients | update_quadrature_points |
70  update_JxW_values);
71 
72  const unsigned int dofs_per_cell = scratch_data.get_n_dofs_per_cell();
73 
74  FullMatrix<double> cell_matrix(dofs_per_cell, dofs_per_cell);
75  Vector<double> cell_rhs(dofs_per_cell);
76 
77  const unsigned int n_q_points = fe_values.get_quadrature().size();
78 
79  std::vector<double> phi_at_q(n_q_points);
80  std::vector<Tensor<1, dim>> grad_phi_at_q(n_q_points, Tensor<1, dim>());
81 
82  std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
83 
84  rhs = 0.0;
85  matrix = 0.0;
86 
87  std::vector<Tensor<1, dim>> a(n_q_points, Tensor<1, dim>());
88 
89  for (const auto &cell : scratch_data.get_dof_handler(this->dof_idx).active_cell_iterators())
90  if (cell->is_locally_owned())
91  {
92  cell_matrix = 0;
93  cell_rhs = 0;
94 
95  fe_values.reinit(cell);
96  fe_values.get_function_values(advected_field_old, phi_at_q);
97  fe_values.get_function_gradients(advected_field_old, grad_phi_at_q);
98 
99  for (int d = 0; d < dim; ++d)
100  {
101  std::vector<double> a_comp(n_q_points);
102  fe_values.get_function_values(advection_velocity.block(d), a_comp);
103  for (const unsigned int q_index : fe_values.quadrature_point_indices())
104  a[q_index][d] = a_comp[q_index];
105  }
106 
107  for (const unsigned int q_index : fe_values.quadrature_point_indices())
108  {
109  for (const unsigned int i : fe_values.dof_indices())
110  {
111  for (const unsigned int j : fe_values.dof_indices())
112  {
113  auto velocity_grad_phi_j = a[q_index] * fe_values.shape_grad(j, q_index);
114  // clang-format off
115  cell_matrix( i, j ) +=
116  ( fe_values.shape_value( i, q_index)
117  *
118  fe_values.shape_value( j, q_index)
119  +
120  data.theta * this->d_tau * ( data.diffusivity *
121  fe_values.shape_grad( i, q_index) *
122  fe_values.shape_grad( j, q_index)
123  +
124  fe_values.shape_value( i, q_index) *
125  velocity_grad_phi_j )
126  ) * fe_values.JxW(q_index);
127  // clang-format on
128  }
129 
130  // clang-format off
131  cell_rhs( i ) +=
132  ( fe_values.shape_value( i, q_index) * phi_at_q[q_index]
133  -
134  ( 1. - data.theta ) * this->d_tau *
135  (
136  data.diffusivity * fe_values.shape_grad( i, q_index) * grad_phi_at_q[q_index]
137  +
138  fe_values.shape_value(i, q_index) * a[q_index] * grad_phi_at_q[q_index]
139  )
140  ) * fe_values.JxW(q_index) ;
141  // clang-format on
142  }
143  } // end gauss
144 
145  // assembly
146  cell->get_dof_indices(local_dof_indices);
147 
148  scratch_data.get_constraint(this->dof_idx)
149  .distribute_local_to_global(cell_matrix, cell_rhs, local_dof_indices, matrix, rhs);
150  }
151 
152  matrix.compress(VectorOperation::add);
153  rhs.compress(VectorOperation::add);
154  }

159  void
160  vmult(VectorType &dst, const VectorType &src) const override
161  {
162  AssertThrow(this->d_tau > 0.0,
163  ExcMessage("advection diffusion operator: d_tau must be set"));
164 
165 
166  scratch_data.get_matrix_free().template cell_loop<VectorType, VectorType>(
167  [&](const auto &matrix_free, auto &dst, const auto &src, auto cell_range) {
168  FECellIntegrator<dim, 1, number> advected_field(matrix_free,
169  this->dof_idx,
170  this->quad_idx);
171  FECellIntegrator<dim, dim, number> velocity(matrix_free, this->dof_idx, this->quad_idx);
172 
173  for (unsigned int cell = cell_range.first; cell < cell_range.second; ++cell)
174  {
175  advected_field.reinit(cell);
176  advected_field.gather_evaluate(src, true, true);
177 
178  velocity.reinit(cell);
179  velocity.read_dof_values_plain(advection_velocity);
180  velocity.evaluate(true, false);
181 
182  for (unsigned int q_index = 0; q_index < advected_field.n_q_points; ++q_index)
183  {
184  const scalar phi = advected_field.get_value(q_index);
185  const vector grad_phi = advected_field.get_gradient(q_index);
186 
187  const scalar velocity_grad_phi =
188  scalar_product(map_to_vector(velocity.get_value(q_index)), grad_phi);
189 
190  advected_field.submit_value(phi + this->d_tau * data.theta * velocity_grad_phi,
191  q_index);
192  advected_field.submit_gradient(this->d_tau * data.theta * data.diffusivity *
193  grad_phi,
194  q_index);
195  }
196  advected_field.integrate_scatter(true, true, dst);
197  }
198  },
199  dst,
200  src,
201  true);
202  }

204  void
205  create_rhs(VectorType &dst, const VectorType &src) const override
206  {
207  /*
208  * This function creates the rhs of the advection-diffusion problem. When inhomogeneous
209  * dirichlet BC are prescribed, the rhs vector is modified including BC terms. Thus the src
210  * vector will NOT be zeroed during the cell_loop.
211  */
212  AssertThrow(this->d_tau > 0.0,
213  ExcMessage("advection diffusion operator: d_tau must be set"));
214 
215  scratch_data.get_matrix_free().template cell_loop<VectorType, VectorType>(
216  [&](const auto &matrix_free, auto &dst, const auto &src, auto macro_cells) {
217  FECellIntegrator<dim, 1, number, VectorizedArrayType> advected_field(matrix_free,
218  this->dof_idx,
219  this->quad_idx);
220  FECellIntegrator<dim, dim, number> velocity(matrix_free, this->dof_idx, this->quad_idx);
221 
222  for (unsigned int cell = macro_cells.first; cell < macro_cells.second; ++cell)
223  {
224  advected_field.reinit(cell);
225  advected_field.gather_evaluate(src, true, true);
226 
227  velocity.reinit(cell);
228  velocity.read_dof_values_plain(advection_velocity);
229  velocity.evaluate(true, false);
230 
231  for (unsigned int q_index = 0; q_index < advected_field.n_q_points; ++q_index)
232  {
233  scalar phi = advected_field.get_value(q_index);
234  const vector grad_phi = advected_field.get_gradient(q_index);
235 
236  const scalar velocity_grad_phi =
237  scalar_product(map_to_vector(velocity.get_value(q_index)), grad_phi);
238  advected_field.submit_value(phi - this->d_tau * (1. - data.theta) *
239  velocity_grad_phi,
240  q_index);
241 
242  advected_field.submit_gradient(-this->d_tau * (1. - data.theta) *
243  data.diffusivity * grad_phi,
244  q_index);
245  }
246 
247  advected_field.integrate_scatter(true, true, dst);
248  }
249  },
250  dst,
251  src,
252  false); // rhs should not be zeroed out in order to consider inhomogeneous dirichlet BC
253  }
