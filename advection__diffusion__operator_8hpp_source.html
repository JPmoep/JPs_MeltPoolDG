50  void
51  assemble_matrixbased(const VectorType &advected_field_old,
52  SparseMatrixType &matrix,
53  VectorType & rhs) const override
54  {
55  AssertThrow(data.diffusivity >= 0.0,
56  ExcMessage("Advection diffusion operator: diffusivity is smaller than zero!"));
57  advected_field_old.update_ghost_values();
58 
59  const auto & mapping = scratch_data.get_mapping();
60  FEValues<dim> fe_values(mapping,
61  scratch_data.get_matrix_free().get_dof_handler(comp).get_fe(),
62  scratch_data.get_matrix_free().get_quadrature(comp),
63  update_values | update_gradients | update_quadrature_points |
64  update_JxW_values);
65  const unsigned int dofs_per_cell = scratch_data.get_n_dofs_per_cell();
66 
67  FullMatrix<double> cell_matrix(dofs_per_cell, dofs_per_cell);
68  Vector<double> cell_rhs(dofs_per_cell);
69 
70  const unsigned int n_q_points = fe_values.get_quadrature().size();
71 
72  std::vector<double> phi_at_q(n_q_points);
73  std::vector<Tensor<1, dim>> grad_phi_at_q(n_q_points, Tensor<1, dim>());
74  std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
75 
76  rhs = 0.0;
77  matrix = 0.0;
78 
79  Tensor<1, dim> a;
80 
81  for (const auto &cell :
82  scratch_data.get_matrix_free().get_dof_handler(comp).active_cell_iterators())
83  if (cell->is_locally_owned())
84  {
85  cell_matrix = 0;
86  cell_rhs = 0;
87 
88  fe_values.reinit(cell);
89  fe_values.get_function_values(advected_field_old, phi_at_q);
90  fe_values.get_function_gradients(advected_field_old, grad_phi_at_q);
91 
92  for (const unsigned int q_index : fe_values.quadrature_point_indices())
93  {
94  auto qCoord = fe_values.get_quadrature_points()[q_index];
95  a = advection_velocity.value(qCoord);
96 
97  for (unsigned int i = 0; i < dofs_per_cell; ++i)
98  {
99  for (unsigned int j = 0; j < dofs_per_cell; ++j)
100  {
101  auto velocity_grad_phi_j = a * fe_values.shape_grad(j, q_index);
102  // clang-format off
103  cell_matrix( i, j ) +=
104  ( fe_values.shape_value( i, q_index)
105  *
106  fe_values.shape_value( j, q_index)
107  +
108  data.theta * this->d_tau * ( data.diffusivity *
109  fe_values.shape_grad( i, q_index) *
110  fe_values.shape_grad( j, q_index) +
111  fe_values.shape_value( i, q_index) ) *
112  velocity_grad_phi_j
113  ) * fe_values.JxW(q_index);
114  // clang-format on
115  }
116 
117  // clang-format off
118  cell_rhs( i ) +=
119  ( fe_values.shape_value( i, q_index) * phi_at_q[q_index]
120  -
121  ( 1. - data.theta ) * this->d_tau *
122  (
123  data.diffusivity * fe_values.shape_grad( i, q_index) * grad_phi_at_q[q_index]
124  +
125  a * grad_phi_at_q[q_index] * fe_values.shape_value( i, q_index)
126  )
127  ) * fe_values.JxW(q_index) ;
128  // clang-format on
129  }
130  } // end gauss
131 
132  // assembly
133  cell->get_dof_indices(local_dof_indices);
134  scratch_data.get_constraint(comp).distribute_local_to_global(
135  cell_matrix, cell_rhs, local_dof_indices, matrix, rhs);
136  }
137 
138  matrix.compress(VectorOperation::add);
139  rhs.compress(VectorOperation::add);
140  }
