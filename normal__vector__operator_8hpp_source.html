45  void
46  assemble_matrixbased(const VectorType &level_set_in,
47  SparseMatrixType &matrix,
48  BlockVectorType & rhs) const override
49  {
50  level_set_in.update_ghost_values();
51  const auto &mapping = scratch_data.get_mapping();
52 
53  FEValues<dim> fe_values(mapping,
54  scratch_data.get_dof_handler(this->dof_idx).get_fe(),
55  scratch_data.get_quadrature(this->quad_idx),
56  update_values | update_gradients | update_quadrature_points |
57  update_JxW_values);
58 
59  const unsigned int dofs_per_cell = scratch_data.get_n_dofs_per_cell(this->dof_idx);
60 
61  FullMatrix<double> normal_cell_matrix(dofs_per_cell, dofs_per_cell);
62  std::vector<Vector<double>> normal_cell_rhs(dim, Vector<double>(dofs_per_cell));
63  std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
64 
65  const unsigned int n_q_points = fe_values.get_quadrature().size();
66 
67  std::vector<Tensor<1, dim>> normal_at_q(n_q_points, Tensor<1, dim>());
68 
69  matrix = 0.0;
70  rhs = 0.0;
71 
72  for (const auto &cell : scratch_data.get_dof_handler(this->dof_idx).active_cell_iterators())
73  if (cell->is_locally_owned())
74  {
75  fe_values.reinit(cell);
76  cell->get_dof_indices(local_dof_indices);
77 
78  normal_cell_matrix = 0.0;
79  for (auto &normal_cell : normal_cell_rhs)
80  normal_cell = 0.0;
81 
82  fe_values.get_function_gradients(
83  level_set_in, normal_at_q); // compute normals from level set solution at tau=0
84  for (const unsigned int q_index : fe_values.quadrature_point_indices())
85  {
86  for (unsigned int i = 0; i < dofs_per_cell; ++i)
87  {
88  const double phi_i = fe_values.shape_value(i, q_index);
89  const Tensor<1, dim> grad_phi_i = fe_values.shape_grad(i, q_index);
90 
91  for (unsigned int j = 0; j < dofs_per_cell; ++j)
92  {
93  const double phi_j = fe_values.shape_value(j, q_index);
94  const Tensor<1, dim> grad_phi_j = fe_values.shape_grad(j, q_index);
95 
96  //clang-format off
97  normal_cell_matrix(i, j) +=
98  (phi_i * phi_j + damping * grad_phi_i * grad_phi_j) *
99  fe_values.JxW(q_index);
100  //clang-format on
101  }
102 
103  for (unsigned int d = 0; d < dim; ++d)
104  {
105  //clang-format off
106  normal_cell_rhs[d](i) +=
107  phi_i * normal_at_q[q_index][d] * fe_values.JxW(q_index);
108  //clang-format on
109  }
110  }
111  }
112 
113  // assembly
114  cell->get_dof_indices(local_dof_indices);
115 
116  scratch_data.get_constraint(this->dof_idx)
117  .distribute_local_to_global(normal_cell_matrix, local_dof_indices, matrix);
118  for (unsigned int d = 0; d < dim; ++d)
119  scratch_data.get_constraint(this->dof_idx)
120  .distribute_local_to_global(normal_cell_rhs[d], local_dof_indices, rhs.block(d));
121 
122  } // end of cell loop
123  matrix.compress(VectorOperation::add);
124  rhs.compress(VectorOperation::add);
125  }

131  void
132  vmult(BlockVectorType &dst, const BlockVectorType &src) const override
133  {
134  scratch_data.get_matrix_free().template cell_loop<BlockVectorType, BlockVectorType>(
135  [&](const auto &, auto &dst, const auto &src, auto cell_range) {
136  FECellIntegrator<dim, dim, number> normal(scratch_data.get_matrix_free(),
137  this->dof_idx,
138  this->quad_idx);
139  for (unsigned int cell = cell_range.first; cell < cell_range.second; ++cell)
140  {
141  normal.reinit(cell);
142  normal.gather_evaluate(src, true, true);
143 
144  for (unsigned int q_index = 0; q_index < normal.n_q_points; ++q_index)
145  {
146  normal.submit_value(normal.get_value(q_index), q_index);
147  normal.submit_gradient(damping * normal.get_gradient(q_index), q_index);
148  }
149 
150  normal.integrate_scatter(true, true, dst);
151  }
152  },
153  dst,
154  src,
155  true);
156  }

158  void
159  create_rhs(BlockVectorType &dst, const VectorType &src) const override
160  {
161  FECellIntegrator<dim, dim, number> normal_vector(scratch_data.get_matrix_free(),
162  this->dof_idx,
163  this->quad_idx);
164  FECellIntegrator<dim, 1, number> level_set(scratch_data.get_matrix_free(),
165  this->dof_idx,
166  this->quad_idx);
167 
168  scratch_data.get_matrix_free().template cell_loop<BlockVectorType, VectorType>(
169  [&](const auto &, auto &dst, const auto &src, auto macro_cells) {
170  for (unsigned int cell = macro_cells.first; cell < macro_cells.second; ++cell)
171  {
172  normal_vector.reinit(cell);
173 
174  level_set.reinit(cell);
175  level_set.read_dof_values_plain(src);
176  level_set.evaluate(false, true);
177 
178  for (unsigned int q_index = 0; q_index < normal_vector.n_q_points; ++q_index)
179  normal_vector.submit_value(level_set.get_gradient(q_index), q_index);
180 
181  normal_vector.integrate_scatter(true, false, dst);
182  }
183  },
184  dst,
185  src,
186  true);
187  }

189  static void
190  get_unit_normals_at_quadrature(const FEValues<dim> & fe_values,
191  const BlockVectorType & normal_vector_field_in,
192  std::vector<Tensor<1, dim>> &unit_normal_at_quadrature)
193  {
194  for (unsigned int d = 0; d < dim; ++d)
195  {
196  std::vector<double> temp(unit_normal_at_quadrature.size());
197  fe_values.get_function_values(normal_vector_field_in.block(d),
198  temp); // compute normals from level set solution at tau=0
199  for (const unsigned int q_index : fe_values.quadrature_point_indices())
200  unit_normal_at_quadrature[q_index][d] = temp[q_index];
201  }
202  for (auto &n : unit_normal_at_quadrature)
203  n /= n.norm(); //@todo: add exception if norm is zero
204  }
