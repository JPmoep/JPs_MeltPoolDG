84  QGauss<dim>( n_q_points ),

86 
87  std::vector<double> phi_at_q( QGauss<dim>( n_q_points ).size() );
88 
89  std::vector<double> volume_fraction;
90  double vol_phase_1 = 0;
91  double vol_phase_2 = 0;
92  const double threshhold = 0.5;
93 
94  for (const auto &cell : dof_handler.active_cell_iterators())
95  if (cell->is_locally_owned())
96  {
97  fe_values.reinit( cell );
98  fe_values.get_function_values( solution_levelset, phi_at_q ); // compute values of old solution
99 
100  for (const unsigned int q_index : fe_values.quadrature_point_indices())
101  {
102  const double phi_normalized = UtilityFunctions::CharacteristicFunctions::normalize ( phi_at_q[q_index], min_value, max_value );
103  if (phi_normalized>=threshhold)
104  vol_phase_1 += fe_values.JxW(q_index);
105  else
106  vol_phase_2 += fe_values.JxW(q_index);
107  }
108  }
109  volume_fraction.emplace_back(Utilities::MPI::sum(vol_phase_1, mpi_communicator));
110  volume_fraction.emplace_back(Utilities::MPI::sum(vol_phase_2, mpi_communicator));
111  //@ todo: write template class for formatted output table
112  if (Utilities::MPI::this_mpi_process(mpi_communicator) == 0)
113  std::cout << "vol phase 1: " << volume_fraction[0] << " vol phase 2: " << volume_fraction[1] << std::endl;
114 
115  volume_table.add_value("time", time);
116  volume_table.add_value("vol phase 1", volume_fraction[0]);
117  volume_table.add_value("vol phase 2", volume_fraction[1]);
